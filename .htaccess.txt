# XAMPP Compatible .htaccess
RewriteEngine On

# Block access to config files
<Files "*.php">
    <RequireAll>
        Require all denied
    </RequireAll>
</Files>

# Allow specific PHP files
<Files "index.php">
    Require all granted
</Files>

<Files "login.php">
    Require all granted  
</Files>

<Files "register.php">
    Require all granted
</Files>

# Block sensitive directories
RewriteRule ^config/ - [F,L]
RewriteRule ^logs/ - [F,L]
RewriteRule ^database/ - [F,L]
